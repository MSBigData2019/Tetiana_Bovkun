Physical_List=['CD','CD Single','Cassette','Vinyl Single', 'LP/EP','DVD Audio','Music Video (Physical)','SACD']
global_physical=global_revenue[global_revenue['Format'].isin(Physical_List)].groupby(['Year'])[["revenue"]].sum()

global_DigitalDown=global_revenue[global_revenue['Format'].isin(DigitalDown_List)].groupby(['Year'])[["revenue"]].sum()
global_DigitalDown['Format']='Digital Download'
